	&=\frac{1}{2\sqrt{x}}

	&=\lim_{h \to 0} \frac{x-(x+h)}{hx(x+h)}\\
	&=\lim_{h \to 0} \frac{-h}{hx(x+h)}\\

\end {flalign*}
